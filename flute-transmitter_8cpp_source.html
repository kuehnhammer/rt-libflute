80 static auto parse_opt(int key, char *arg, struct argp_state *state) -> error_t {
81  auto arguments = static_cast<struct ft_arguments *>(state->input);
82  switch (key) {
83  case 'm':
84  arguments->mcast_target = arg;
85  break;
86  case 'k':
87  arguments->aes_key = arg;
88  arguments->enable_ipsec = true;
89  break;
90  case 'p':
91  arguments->mcast_port = static_cast<unsigned short>(strtoul(arg, nullptr, 10));
92  break;
93  case 't':
94  arguments->mtu = static_cast<unsigned short>(strtoul(arg, nullptr, 10));
95  break;
96  case 'r':
97  arguments->rate_limit = static_cast<uint32_t>(strtoul(arg, nullptr, 10));
98  break;
99  case 'l':
100  arguments->log_level = static_cast<unsigned>(strtoul(arg, nullptr, 10));
101  break;
102  case 'f':
103  arguments->fec = static_cast<unsigned>(strtoul(arg, nullptr, 10));
104  if ( (arguments->fec | 1) != 1 ) {
105  spdlog::error("Invalid FEC scheme ! Please pick either 0 (Compact No Code) or 1 (Raptor)");
106  return ARGP_ERR_UNKNOWN;
107  }
108  break;
109  case ARGP_KEY_NO_ARGS:
110  argp_usage (state);
111  case ARGP_KEY_ARG:
112  arguments->files = &state->argv[state->next-1];
113  state->next = state->argc;
114  break;
115  default:
116  return ARGP_ERR_UNKNOWN;
117  }
118  return 0;
119 }

141 auto main(int argc, char **argv) -> int {
142  struct ft_arguments arguments;
143  /* Default values */
144  arguments.mcast_target = "238.1.1.95";
145 
146  argp_parse(&argp, argc, argv, 0, nullptr, &arguments);
147 
148  // Set up logging
149  std::string ident = "flute-transmitter";
150  auto syslog_logger = spdlog::syslog_logger_mt("syslog", ident, LOG_PID | LOG_PERROR | LOG_CONS );
151 
152  spdlog::set_level(
153  static_cast<spdlog::level::level_enum>(arguments.log_level));
154  spdlog::set_pattern("[%H:%M:%S.%f %z] [%^%l%$] [thr %t] %v");
155 
156  spdlog::set_default_logger(syslog_logger);
157  spdlog::info("FLUTE transmitter demo starting up");
158 
159  try {
160  // We're responsible for buffer management, so create a vector of structs that
161  // are going to hold the data buffers
162  struct FsFile {
163  std::string location;
164  char* buffer;
165  size_t len;
166  uint32_t toi;
167  };
168  std::vector<FsFile> files;
169 
170  // read the file contents into the buffers
171  for (int j = 0; arguments.files[j]; j++) {
172  struct stat sb;
173  int fd;
174  fd = open(arguments.files[j], O_RDONLY);
175  if (fd == -1) {
176  spdlog::error("Couldnt open file {}",arguments.files[j]);
177  continue;
178  }
179  if (fstat(fd, &sb) == -1){ // To obtain file size
180  spdlog::error("fstat() call for file {} failed",arguments.files[j]);
181  close(fd);
182  continue;
183  }
184  if (sb.st_size <= 0) {
185  close(fd);
186  continue;
187  }
188  char* buffer = (char*) mmap(nullptr, sb.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
189  close(fd);
190  if ( (long) buffer <= 0) {
191  spdlog::error("mmap() failed for file {}",arguments.files[j]);
192  continue;
193  }
194  files.push_back(FsFile{ arguments.files[j], buffer, (size_t) sb.st_size});
195  }
196 
197  // Create a Boost io_service
198  boost::asio::io_service io;
199 
200  // Construct the transmitter class
201  LibFlute::Transmitter transmitter(
202  arguments.mcast_target,
203  (short)arguments.mcast_port,
204  16,
205  arguments.mtu,
206  arguments.rate_limit,
207  LibFlute::FecScheme(arguments.fec),
208  io);
209 
210  // Configure IPSEC ESP, if enabled
211  if (arguments.enable_ipsec)
212  {
213  transmitter.enable_ipsec(1, arguments.aes_key);
214  }
215 
216  // Register a completion callback
217  transmitter.register_completion_callback(
218  [&files](uint32_t toi) {
219  for (auto& file : files) {
220  if (file.toi == toi) {
221  spdlog::info("{} (TOI {}) has been transmitted", file.location,file.toi);
222  munmap(file.buffer,file.len);
223  }
224  }
225  });
226 
227  // Queue all the files
228  for (auto& file : files) {
229  file.toi = transmitter.send( file.location,
230  "application/octet-stream",
231  transmitter.seconds_since_epoch() + 60, // 1 minute from now
232  file.buffer,
233  file.len
234  );
235  if (file.toi > 0) {
236  spdlog::info("Queued {} ({} bytes) for transmission, TOI is {}",
237  file.location, file.len, file.toi);
238  }
239  }
240 
241  // Start the io_service, and thus sending data
242  io.run();
243  } catch (std::exception ex ) {
244  spdlog::error("Exiting on unhandled exception: %s", ex.what());
245  }
246 
247 exit:
248  return 0;
249 }
