66 auto LibFlute::PcapReceiver::process_packet() -> void
67 {
68  assert(_packet_data != nullptr);
69 
70  // Check if the destination matches the mcast address and port we need and
71  // pass the payload on for FLUTE decoding if it does
72  check_packet();
73 
74  // read the next packet
75  read_packet();
76 
77  if (_packet_data == nullptr) {
78  spdlog::info("Last packet processed, exiting.");
79  } else {
80  // Calculate how long we have to wait until processing the next packet
81  auto packet_time = tv_to_usecs(&_packet_header.ts);
82  auto delta = packet_time - _last_packet_time;
83  _last_packet_time = packet_time;
84 
85  _packet_timer.expires_from_now(boost::posix_time::microseconds(delta));
86  _packet_timer.async_wait( boost::bind(&PcapReceiver::process_packet, this)); //NOLINT
87  }
88 }

96 auto LibFlute::PcapReceiver::check_packet() -> void
97 {
98  assert(_packet_data != nullptr);
99 
100  auto ip_header = (struct ip*)(_packet_data);
101  auto udp_header = (struct udphdr*)(_packet_data + (ip_header->ip_hl * 4));
102 
103  auto dest_address = std::string(inet_ntoa(ip_header->ip_dst));
104  auto dest_port = ntohs(udp_header->uh_dport);
105 
106  if (dest_address == _mcast_address && dest_port == _mcast_port) {
107  auto payload = (_packet_data + (ip_header->ip_hl * 4) + sizeof(struct udphdr));
108  handle_received_packet((char*)payload, ntohs(udp_header->uh_ulen) - sizeof(struct udphdr));
109  }
110 }
