24 LibFlute::AlcPacket::AlcPacket(char* data, size_t len)
25 {
26  if (len < 4) {
27  throw "Packet too short";
28  }
29 
30  std::memcpy(&_lct_header, data, 4);
31  if (_lct_header.version != 1) {
32  throw "Unsupported LCT version";
33  }
34 
35  char* hdr_ptr = data + 4;
36  if (_lct_header.congestion_control_flag != 0) {
37  throw "Unsupported CCI field length";
38  }
39  // [TODO] read CCI
40  hdr_ptr += 4;
41 
42  if (_lct_header.half_word_flag == 0 && _lct_header.tsi_flag == 0) {
43  throw "TSI field not present";
44  }
45  auto tsi_shift = 0;
46  if(_lct_header.half_word_flag == 1) {
47  _tsi = ntohs(*(uint16_t*)hdr_ptr);
48  tsi_shift = 16;
49  hdr_ptr += 2;
50  }
51  if(_lct_header.tsi_flag == 1) {
52  _tsi |= ntohl(*(uint32_t*)hdr_ptr) << tsi_shift;
53  hdr_ptr += 4;
54  }
55 
56  if ( _lct_header.close_session_flag == 0 && _lct_header.half_word_flag == 0 && _lct_header.toi_flag == 0) {
57  throw "TOI field not present";
58  }
59  auto toi_shift = 0;
60  if(_lct_header.half_word_flag == 1) {
61  _toi = ntohs(*(uint16_t*)hdr_ptr);
62  toi_shift = 16;
63  hdr_ptr += 2;
64  }
65  switch(_lct_header.toi_flag) {
66  case 0: break;
67  case 1:
68  _toi |= ntohl(*(uint32_t*)hdr_ptr) << toi_shift;
69  hdr_ptr += 4;
70  break;
71  case 2:
72  if (toi_shift > 0) {
73  throw "TOI fields over 64 bits in length are not supported";
74  } else {
75  _toi = ntohl(*(uint32_t*)hdr_ptr);
76  hdr_ptr += 4;
77  _toi |= (uint64_t)(ntohl(*(uint32_t*)hdr_ptr)) << 32;
78  hdr_ptr += 4;
79  }
80  break;
81  default:
82  throw "TOI fields over 64 bits in length are not supported";
83  }
84 
85  switch (_lct_header.codepoint) {
86  case 0:
87  _fec_oti.encoding_id = FecScheme::CompactNoCode;
88  break;
89  case 1:
90  _fec_oti.encoding_id = FecScheme::Raptor;
91  break;
92  default:
93  throw "Only the Compact No-Code and Raptor FEC schemes are supported";
94  break;
95  }
96 
97  auto expected_header_len = 2 +
98  _lct_header.congestion_control_flag +
99  _lct_header.half_word_flag +
100  _lct_header.tsi_flag +
101  _lct_header.toi_flag;
102 
103  auto ext_header_len = (_lct_header.lct_header_len - expected_header_len) * 4;
104 
105  while (ext_header_len > 0) {
106  uint8_t het = *hdr_ptr;
107  hdr_ptr += 1;
108  uint8_t hel = 0;
109  if (het < 128) {
110  hel = *hdr_ptr;
111  hdr_ptr += 1;
112  }
113 
114  switch ((AlcPacket::HeaderExtension)het) {
115  case EXT_NOP:
116  case EXT_AUTH:
117  case EXT_TIME: {
118  hdr_ptr += 3;
119  break; // ignored
120  }
121  case EXT_FTI: {
122  switch (_fec_oti.encoding_id) {
123  case FecScheme::CompactNoCode:
124  if (hel != 4) {
125  throw "Invalid length for EXT_FTI header extension for Compact No Code FEC scheme";
126  }
127  _fec_oti.transfer_length = (uint64_t)(ntohs(*(uint16_t*)hdr_ptr)) << 32;
128  hdr_ptr += 2;
129  _fec_oti.transfer_length |= (uint64_t)(ntohl(*(uint32_t*)hdr_ptr));
130  hdr_ptr += 4;
131  hdr_ptr += 2; // reserved
132  _fec_oti.encoding_symbol_length = ntohs(*(uint16_t*)hdr_ptr);
133  hdr_ptr += 2;
134  _fec_oti.max_source_block_length = ntohl(*(uint32_t*)hdr_ptr);
135  hdr_ptr += 4;
136  break;
137  case FecScheme::Raptor:
138  //TODO
139  spdlog::warn("Raptor FEC support in EXT_FTI header extension is still in progress");
140  throw "Raptor FEC support in EXT_FTI header extension is still in progress";
141  break;
142  default:
143  throw "Unsupported FEC scheme";
144  break;
145  }
146  break;
147  }
148  case EXT_FDT: {
149  uint8_t flute_version = (*hdr_ptr & 0xF0) >> 4;
150  if (flute_version != 1) {
151  throw "Only FLUTE version 1 is supported";
152  }
153  _fdt_instance_id = (*hdr_ptr & 0x0F) << 16;
154  hdr_ptr++;
155  _fdt_instance_id |= ntohs(*(uint16_t*)hdr_ptr);
156  hdr_ptr += 2;
157  break;
158  }
159  case EXT_CENC: {
160  uint8_t encoding = *hdr_ptr;
161  switch (encoding) {
162  case 0: _content_encoding = ContentEncoding::NONE; break;
163  case 1: _content_encoding = ContentEncoding::ZLIB; break;
164  case 2: _content_encoding = ContentEncoding::DEFLATE; break;
165  case 3: _content_encoding = ContentEncoding::GZIP; break;
166  }
167  hdr_ptr += 3;
168  break;
169  }
170  }
171 
172  ext_header_len -= 4;
173  ext_header_len -= hel * 4;
174  }
175 }

177 LibFlute::AlcPacket::AlcPacket(uint16_t tsi, uint16_t toi, LibFlute::FecOti fec_oti, const std::vector<LibFlute::EncodingSymbol>& symbols, size_t max_size, uint32_t fdt_instance_id) // NOLINT
178  : _fec_oti(std::move(fec_oti))
179 {
180  auto lct_header_len = 3;
181  if (toi == 0) { // Add extensions for FDT
182  lct_header_len += 5;
183  }
184 
185  auto max_packet_length = max_size +
186  static_cast<long>(lct_header_len) * 4
187  + 4 ;
188 
189  _buffer = (char*)calloc(max_packet_length, sizeof(char));
190 
191  auto* lct_header = (lct_header_t*)_buffer;
192 
193  lct_header->version = 1;
194  lct_header->half_word_flag = 1;
195  if (_fec_oti.encoding_id == LibFlute::FecScheme::CompactNoCode) {
196  lct_header->codepoint = 0;
197  } else if (_fec_oti.encoding_id == LibFlute::FecScheme::Raptor) {
198  lct_header->codepoint = 1;
199  } else {
200  throw "Unsupported FEC scheme";
201  }
202  lct_header->lct_header_len = lct_header_len;
203  lct_header->codepoint = (uint8_t)_fec_oti.encoding_id;
204  auto* hdr_ptr = _buffer + 4;
205  auto* payload_ptr = _buffer + 4UL * lct_header_len;
206 
207  auto payload_size = EncodingSymbol::to_payload(symbols, payload_ptr, max_size, _fec_oti);
208  _len = 4L * lct_header_len + payload_size;
209 
210  hdr_ptr += 4; // CCI = 0
211 
212  *((uint16_t*)hdr_ptr) = htons(tsi);
213  hdr_ptr += 2;
214 
215  *((uint16_t*)hdr_ptr) = htons(toi);
216  hdr_ptr += 2;
217 
218  if (toi == 0) { // Add extensions for FDT
219  *((uint8_t*)hdr_ptr) = EXT_FDT;
220  hdr_ptr += 1;
221  *((uint8_t*)hdr_ptr) = 1 << 4 | (fdt_instance_id & 0x000F0000) >> 16;
222  hdr_ptr += 1;
223  *((uint16_t*)hdr_ptr) = htons(fdt_instance_id & 0x0000FFFF);
224  hdr_ptr += 2;
225 
226  *((uint8_t*)hdr_ptr) = EXT_FTI;
227  hdr_ptr += 1;
228  *((uint8_t*)hdr_ptr) = 4; // HEL
229  hdr_ptr += 1;
230  *((uint16_t*)hdr_ptr) = htons((_fec_oti.transfer_length & 0x00FF0000) >> 32);
231  hdr_ptr += 2;
232  *((uint32_t*)hdr_ptr) = htonl(_fec_oti.transfer_length & 0x0000FFFF);
233  hdr_ptr += 4;
234  hdr_ptr += 2; // reserved
235  *((uint16_t*)hdr_ptr) = htons(_fec_oti.encoding_symbol_length);
236  hdr_ptr += 2;
237  *((uint32_t*)hdr_ptr) = htonl(_fec_oti.max_source_block_length);
238  }
239 }
