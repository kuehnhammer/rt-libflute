193 auto LibFlute::FileDeliveryTable::to_string() const -> std::string {
194  tinyxml2::XMLDocument doc;
195  doc.InsertFirstChild( doc.NewDeclaration() );
196  auto* root = doc.NewElement("FDT-Instance");
197  root->SetAttribute("Expires", std::to_string(_expires).c_str());
198  root->SetAttribute("FEC-OTI-FEC-Encoding-ID", (unsigned)_global_fec_oti.encoding_id);
199  root->SetAttribute("FEC-OTI-Maximum-Source-Block-Length", (unsigned)_global_fec_oti.max_source_block_length);
200  root->SetAttribute("FEC-OTI-Encoding-Symbol-Length", (unsigned)_global_fec_oti.encoding_symbol_length);
201  root->SetAttribute("xmlns:mbms2007", "urn:3GPP:metadata:2007:MBMS:FLUTE:FDT");
202  doc.InsertEndChild(root);
203 
204  for (const auto& file : _file_entries) {
205  auto* f = doc.NewElement("File");
206  f->SetAttribute("TOI", file.toi);
207  f->SetAttribute("Content-Location", file.content_location.c_str());
208  f->SetAttribute("Content-Length", file.content_length);
209  f->SetAttribute("Transfer-Length", (unsigned)file.fec_oti.transfer_length);
210  f->SetAttribute("Content-MD5", file.content_md5.c_str());
211  f->SetAttribute("Content-Type", file.content_type.c_str());
212  if(file.fec_transformer) {
213  file.fec_transformer->add_fdt_info(f);
214  }
215  auto* cc = doc.NewElement("mbms2007:Cache-Control");
216  auto* exp = doc.NewElement("mbms2007:Expires");
217  exp->SetText(std::to_string(file.expires).c_str());
218  cc->InsertEndChild(exp);
219  f->InsertEndChild(cc);
220  root->InsertEndChild(f);
221  }
222 
223 
224  tinyxml2::XMLPrinter printer;
225  doc.Print(&printer);
226  return {printer.CStr()};
227 }
