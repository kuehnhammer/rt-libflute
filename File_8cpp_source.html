138 auto LibFlute::File::put_symbol( const LibFlute::EncodingSymbol& symbol ) -> void
139 {
140  if(_complete) {
141  spdlog::debug("Not handling symbol {} , SBN {} since file is already complete",symbol.id(),symbol.source_block_number());
142  return;
143  }
144  if (symbol.source_block_number() > _source_blocks.size()) {
145  throw "Source Block number too high";
146  }
147 
148  SourceBlock& source_block = _source_blocks[ symbol.source_block_number() ];
149 
150  if(source_block.complete){
151  spdlog::warn("Ignoring symbol {} since block {} is already complete",symbol.id(),symbol.source_block_number());
152  return;
153  }
154 
155  if (symbol.id() > source_block.symbols.size()) {
156  throw "Encoding Symbol ID too high";
157  }
158 
159  LibFlute::Symbol& target_symbol = source_block.symbols[symbol.id()];
160 
161  if (!target_symbol.complete) {
162  symbol.decode_to(target_symbol.data, target_symbol.length);
163  target_symbol.complete = true;
164  if (_meta.fec_transformer) {
165  _meta.fec_transformer->process_symbol(source_block,target_symbol,symbol.id());
166  }
167  check_source_block_completion(source_block);
168  check_file_completion();
169  }

193 auto LibFlute::File::calculate_partitioning() -> void
194 {
195  if (_meta.fec_transformer && _meta.fec_transformer->calculate_partitioning()){
196  _nof_source_symbols = _meta.fec_transformer->nof_source_symbols;
197  _nof_source_blocks = _meta.fec_transformer->nof_source_blocks;
198  _large_source_block_length = _meta.fec_transformer->large_source_block_length;
199  _small_source_block_length = _meta.fec_transformer->small_source_block_length;
200  _nof_large_source_blocks = _meta.fec_transformer->nof_large_source_blocks;
201  return;
202  }
203  // Calculate source block partitioning (RFC5052 9.1)
204  _nof_source_symbols = ceil((double)_meta.fec_oti.transfer_length / (double)_meta.fec_oti.encoding_symbol_length);
205  _nof_source_blocks = ceil((double)_nof_source_symbols / (double)_meta.fec_oti.max_source_block_length);
206  _large_source_block_length = ceil((double)_nof_source_symbols / (double)_nof_source_blocks);
207  _small_source_block_length = floor((double)_nof_source_symbols / (double)_nof_source_blocks);
208  _nof_large_source_blocks = _nof_source_symbols - _small_source_block_length * _nof_source_blocks;
209 }

211 auto LibFlute::File::create_blocks() -> void
212 {
213  // Create the required source blocks and encoding symbols
214 
215  if (_meta.fec_transformer){
216  int bytes_read = 0;
217  _source_blocks = _meta.fec_transformer->create_blocks(_buffer, &bytes_read);
218  if (_source_blocks.empty()) {
219  spdlog::error("FEC Transformer failed to create source blocks");
220  throw "FEC Transformer failed to create source blocks";
221  }
222  return;
223  }
224 
225  auto* buffer_ptr = _buffer;
226  size_t remaining_size = _meta.fec_oti.transfer_length;
227  auto number = 0;
228  while (remaining_size > 0) {
229  LibFlute::SourceBlock block;
230  auto symbol_id = 0;
231  auto block_length = ( number < _nof_large_source_blocks ) ? _large_source_block_length : _small_source_block_length;
232 
233  for (int i = 0; i < block_length; i++) {
234  auto symbol_length = std::min(remaining_size, (size_t)_meta.fec_oti.encoding_symbol_length);
235  assert(buffer_ptr + symbol_length <= _buffer + _meta.fec_oti.transfer_length);
236 
237  LibFlute::Symbol symbol{ .data = buffer_ptr, .length = symbol_length, .complete = false};
238  block.symbols[ symbol_id++ ] = symbol;
239 
240  remaining_size -= symbol_length;
241  buffer_ptr += symbol_length;
242 
243  if (remaining_size <= 0) {
244  break;
245  }
246  }
247  _source_blocks[number++] = block;
248  }
249 }

251 auto LibFlute::File::get_next_symbols(size_t max_size) -> std::vector<EncodingSymbol>
252 {
253  int nof_symbols = std::floor((float)max_size / (float)_meta.fec_oti.encoding_symbol_length);
254  auto cnt = 0;
255  std::vector<EncodingSymbol> symbols;
256  spdlog::debug("Attempting to queue {} symbols",nof_symbols);
257  for (auto& block : _source_blocks) {
258  if (cnt >= nof_symbols) {
259  break;
260  }
261 
262  if (!block.second.complete) {
263  for (auto& symbol : block.second.symbols) {
264  if (cnt >= nof_symbols) {
265  break;
266  }
267 
268  if (!symbol.second.complete && !symbol.second.queued) {
269  symbols.emplace_back(symbol.first, block.first, symbol.second.data, symbol.second.length, _meta.fec_oti.encoding_id);
270  symbol.second.queued = true;
271  cnt++;
272  }
273  }
274  }
275  }
276  return symbols;
277

296 auto LibFlute::calculate_md5(char *input, size_t length, unsigned char *result) -> unsigned int
297 {
298  // simple implementation based on openssl docs (https://www.openssl.org/docs/man3.0/man3/EVP_DigestInit_ex.html)
299  if (input == nullptr || length == 0U) {
300  spdlog::error("MD5 called with invalid input");
301  return -1;
302  }
303  spdlog::debug("MD5 calculation called for input length {}", length);
304 
305  EVP_MD_CTX* context = EVP_MD_CTX_new();
306  const EVP_MD* md = EVP_md5();
307  unsigned int md_len;
308 
309  EVP_DigestInit_ex(context, md, nullptr);
310  EVP_DigestUpdate(context, input, length);
311  EVP_DigestFinal_ex(context, result, &md_len);
312  EVP_MD_CTX_free(context);
313 
314  char buf [EVP_MAX_MD_SIZE * 2] = {}; //NOLINT
315  for (auto i = 0UL; i < md_len; i++){
316  sprintf(&buf[i*2], "%02x", result[i]);
317  }
318  spdlog::debug("MD5 Digest is {}", buf);
319 
320  return md_len;
321 }
