83 static auto parse_opt(int key, char *arg, struct argp_state *state) -> error_t {
84  auto arguments = static_cast<struct ft_arguments *>(state->input);
85  switch (key) {
86  case 'c':
87  arguments->capture_file = arg;
88  break;
89  case 'm':
90  arguments->mcast_target = arg;
91  break;
92  case 'i':
93  arguments->flute_interface = arg;
94  break;
95  case 'k':
96  arguments->aes_key = arg;
97  arguments->enable_ipsec = true;
98  break;
99  case 'p':
100  arguments->mcast_port = static_cast<unsigned short>(strtoul(arg, nullptr, 10));
101  break;
102  case 'l':
103  arguments->log_level = static_cast<unsigned>(strtoul(arg, nullptr, 10));
104  break;
105  case 'd':
106  arguments->download_dir = arg;
107  break;
108  case 'n':

113  break;
114  default:
115  return ARGP_ERR_UNKNOWN;

141 auto main(int argc, char **argv) -> int {
142  struct ft_arguments arguments;
143  /* Default values */
144  arguments.mcast_target = "238.1.1.95";
145  arguments.flute_interface= "0.0.0.0";
146 
147  // Parse the arguments
148  argp_parse(&argp, argc, argv, 0, nullptr, &arguments);
149 
150  // Set up logging
151  std::string ident = "flute-receiver";
152  auto syslog_logger = spdlog::syslog_logger_mt("syslog", ident, LOG_PID | LOG_PERROR | LOG_CONS );
153 
154  spdlog::set_level(
155  static_cast<spdlog::level::level_enum>(arguments.log_level));
156  spdlog::set_pattern("[%H:%M:%S.%f %z] [%^%l%$] [thr %t] %v");
157 
158  spdlog::set_default_logger(syslog_logger);
159  spdlog::info("FLUTE receiver demo starting up");
160 
161  try {
162  // Create a Boost io_service
163  boost::asio::io_service io;
164 
165  std::shared_ptr<LibFlute::ReceiverBase> receiver;
166 
167  // Create the receiver
168  if (arguments.capture_file != nullptr) {
169  try {
170  receiver = std::make_shared<LibFlute::PcapReceiver>(
171  arguments.capture_file,
172  arguments.mcast_target,
173  arguments.mcast_port,
174  arguments.tsi,
175  io);
176  } catch (std::runtime_error& ex) {
177  spdlog::error("PCAP receiver error. {}", ex.what());
178  exit(1);
179  }
180  } else {
181  auto net_receiver = std::make_shared<LibFlute::Receiver>(
182  arguments.flute_interface,
183  arguments.mcast_target,
184  arguments.mcast_port,
185  arguments.tsi,
186  io);
187 
188  // Configure IPSEC, if enabled
189  if (arguments.enable_ipsec)
190  {
191  net_receiver->enable_ipsec(1, arguments.aes_key);
192  }
193 
194  receiver = net_receiver;
195  }
196 
197  receiver->register_completion_callback(
198  [&](std::shared_ptr<LibFlute::File> file) { //NOLINT
199  spdlog::info("{} (TOI {}) has been received",
200  file->meta().content_location, file->meta().toi);
201  char *buf = (char*) calloc(256,1);
202  char *fname = (char*) strrchr(file->meta().content_location.c_str(),'/');
203  if(!fname){
204  fname = (char*) file->meta().content_location.c_str();
205  } else {
206  fname++;
207  }
208  if (arguments.download_dir) {
209  snprintf(buf,256,"%s/%s",arguments.download_dir, fname);
210  } else {
211  snprintf(buf,256,"flute_download_%d-%s",file->meta().toi, fname);
212  }
213  FILE* fd = fopen(buf, "wb");
214  if (fd) {
215  fwrite(file->buffer(), 1, file->length(), fd);
216  fclose(fd);
217  } else {
218  spdlog::error("Error opening file {} to store received object",buf);
219  }
220  free(buf);
221  if (file->meta().toi == arguments.nfiles) {
222  spdlog::warn("{} file(s) received. Stopping reception",arguments.nfiles);
223  receiver->stop();
224  }
225  });
226 
227  // Start the IO service
228  io.run();
229  } catch (std::exception ex ) {
230  spdlog::error("Exiting on unhandled exception: {}", ex.what());
231  }
232 
233 exit:
234  return 0;
235 }
