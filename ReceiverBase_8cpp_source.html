43 auto LibFlute::ReceiverBase::handle_received_packet(char* data, size_t bytes) -> void
44 {
45  spdlog::info("processing {} bytes", bytes);
46  try {
47  auto alc = LibFlute::AlcPacket(data, bytes);
48 
49  if (alc.tsi() != _tsi) {
50  spdlog::warn("Discarding packet for unknown TSI {}", alc.tsi());
51  return;
52  }
53 
54  const std::lock_guard<std::mutex> lock(_files_mutex);
55 
56  if (alc.toi() == 0 && (!_fdt || _fdt->instance_id() != alc.fdt_instance_id())) {
57  if (_files.find(alc.toi()) == _files.end()) {
58  FileDeliveryTable::FileEntry fe{0, "", static_cast<uint32_t>(alc.fec_oti().transfer_length), "", "", 0, alc.fec_oti(), nullptr};
59  _files.emplace(alc.toi(), std::make_shared<LibFlute::File>(fe));
60  }
61  }
62 
63  if (_files.find(alc.toi()) != _files.end() && !_files[alc.toi()]->complete()) {
64  auto encoding_symbols = LibFlute::EncodingSymbol::from_payload(
65  data + alc.header_length(),
66  bytes - alc.header_length(),
67  _files[alc.toi()]->fec_oti(),
68  alc.content_encoding());
69 
70  for (const auto& symbol : encoding_symbols) {
71 
72  spdlog::debug("received TOI {} SBN {} ID {}", alc.toi(), symbol.source_block_number(), symbol.id() );
73  _files[alc.toi()]->put_symbol(symbol);
74  }
75 
76  auto* file = _files[alc.toi()].get();
77  if (_files[alc.toi()]->complete()) {
78  for (auto it = _files.begin(); it != _files.end();)
79  {
80  if (it->second.get() != file && it->second->meta().content_location == file->meta().content_location)
81  {
82  spdlog::debug("Replacing file with TOI {}", it->first);
83  it = _files.erase(it);
84  }
85  else
86  {
87  ++it;
88  }
89  }
90 
91  spdlog::debug("File with TOI {} completed", alc.toi());
92  if (alc.toi() != 0 && _completion_cb) {
93  _completion_cb(_files[alc.toi()]);
94  _files.erase(alc.toi());
95  }
96 
97  if (alc.toi() == 0) { // parse complete FDT
98  _fdt = std::make_unique<LibFlute::FileDeliveryTable>(
99  alc.fdt_instance_id(), _files[alc.toi()]->buffer(), _files[alc.toi()]->length());
100 
101  _files.erase(alc.toi());
102  for (const auto& file_entry : _fdt->file_entries()) {
103  // automatically receive all files in the FDT
104  if (_files.find(file_entry.toi) == _files.end()) {
105  spdlog::debug("Starting reception for file with TOI {}: {} ({})", file_entry.toi,
106  file_entry.content_location, file_entry.content_type);
107  _files.emplace(file_entry.toi, std::make_shared<LibFlute::File>(file_entry));
108  }
109  }
110  }
111  }
112  } else {
113  spdlog::trace("Discarding packet for unknown or already completed file with TOI {}", alc.toi());
114  }
115  } catch (std::exception& ex) {
116  spdlog::warn("Failed to decode ALC/FLUTE packet: {}", ex.what());
117  }
118 }

129 auto LibFlute::ReceiverBase::remove_expired_files(unsigned max_age) -> void
130 {
131  const std::lock_guard<std::mutex> lock(_files_mutex);
132  for (auto it = _files.cbegin(); it != _files.cend();)
133  {
134  auto age = time(nullptr) - it->second->received_at();
135  if ( it->second->meta().content_location != "bootstrap.multipart" && age > max_age) {
136  it = _files.erase(it);
137  } else {

139  }
